
# 유니온 파인드
### 노드의 합집합을 판별하는 알고리즘

### return parent[now] = find(parent[now])를 통해 
###부모를 계속 갱신하여 시간을 줄여야하는 것을 잊지 말자

##
    public static int find(int now) {
        if (parent[now] == now) {
            return now;
        }
        return parent[now] = find(parent[now]);
    }

    public static void union(int a, int b) {
        a = find(a);
        b = find(b);
        if (a < b) {
            parent[b] = a;
        } else {
            parent[a] = b;
        }
    }
